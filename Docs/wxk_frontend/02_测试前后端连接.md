

测试前后端连接代码（保存为`test.html`用浏览器打开即可）

```html
<!DOCTYPE html>
<html lang="zh">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>FastAPI 测试</title>
</head>
<body>
    <h1>FastAPI 连接测试</h1>
    <button onclick="testConnection()">测试连接</button>
    <p id="result"></p>

    <script>
        async function testConnection() {
            const apiUrl = "https://exact-relieved-yak.ngrok-free.app/ping"; // 你的后端 API 地址
            try {
                const response = await fetch(apiUrl, {
                    method: "GET",
                    headers: {
                        "ngrok-skip-browser-warning": "true" // 关键代码
                    }
                });
                const data = await response.json();
                document.getElementById("result").innerText = "服务器返回: " + JSON.stringify(data);
            } catch (error) {
                document.getElementById("result").innerText = "请求失败: " + error;
            }
        }
    </script>
    
</body>
</html>

```
